👤 创作者端操作路径

登录 → 提交作品 → 查看积分 → 兑换奖励
flowchart TD
    START(["🌐 打开创作者端"]) --> LOGIN{"已有账号?"}
    LOGIN -->|"🆕 新创作者"| REG["填写注册信息
Discord / Account ID / Game UID
姓名 / 服务器 / 地址 / 电话"]
    LOGIN -->|"✅ 已注册"| AUTH["输入 Discord + UID 验证"]
    REG --> AUTH
    AUTH --> DASH["🎯 Dashboard 主页
5 个功能 Tab"]
    DASH --> TAB1["📝 Submit Works"]
    TAB1 --> CHECK{"剩余提交次数?"}
    CHECK -->|"≥1 次"| FORM["填写提交表单
统一链接框 (图片/视频混投)
反馈 (选填)"]
    CHECK -->|"0 次"| LOCKED["🔒 已用完机会
表单变灰"]
    FORM --> SUBMIT["✅ 提交成功
显示第 X/2 次"]
    SUBMIT --> TAB1
    DASH --> TAB2["📋 My Submissions"]
    TAB2 --> STATS["📊 本月统计数据
已提交 X 条 · 已评分/待审核"]
    STATS --> CC{"≥15条?"}
    CC -->|"是"| CC_DONE["🎁 CC达标"]
    CC -->|"否"| CC_PROGRESS["📊 进度 X/15"]
    STATS --> TIER{"检测等级"}
    TIER -->|"认证"| UPGRADE["连续2月≥5分 → 金级"]
    TIER -->|"金级"| UPGRADE2["连续3月≥5分 → 铂金"]
    TIER -->|"铂金"| TOP["🏆 最高等级"]
    DASH --> TAB3["🎁 Redeem Rewards"]
    TAB3 --> CHECK2{"开放兑换?"}
    CHECK2 -->|"否"| LOCKED2["🔒 暂不开放"]
    CHECK2 -->|"是"| REWARDS["💎 钻石 / 🎮 谷歌卡 / 📦 周边"]
    REWARDS --> CONFIRM["✅ 确认兑换"]
    DASH --> TAB4["❓ FAQ"]
    TAB4 --> SEARCH["🔍 搜索"]
    TAB4 --> POST["📝 Post in Discord"]

⚠️ 投稿数据抓取失败

AI Agent 扫描各平台链接时的异常处理
flowchart TD
    START["🤖 Agent 扫描链接"] --> TRY["打开链接"]
    TRY --> RESULT{"页面可访问?"}
    RESULT -->|"✅ 正常"| PARSE["解析互动数/播放量"]
    RESULT -->|"❌ 404"| LOG1["记录: 不可访问"]
    RESULT -->|"❌ 需登录"| LOG2["记录: 平台限制"]
    RESULT -->|"❌ 超时"| LOG3["记录: 超时"]
    LOG1 --> RETRY{"重试<3次?"}
    LOG2 --> RETRY
    LOG3 --> RETRY
    RETRY -->|"是"| TRY
    RETRY -->|"否"| FAILED["标记抓取失败"]
    FAILED --> REVIEW["管理员手动补充"]
    PARSE --> DONE["写入算分 JSON"]
